## CODE

```python
"""
패턴 찾기 - 시계 방향 
방향 전환 조건: 벽이거나, 이미 숫자가 채워진 칸 
"""
import sys

sys.stdin = open('input.txt', 'r')

T = int(input().strip())

for t in range(1, T+1):
    N = int(input().strip())

    arr = [[0]*N for _ in range(N)]

    # 오른쪽, 아래, 왼쪽, 위 
    dr = [0, 1, 0, -1]
    dc = [1, 0, -1, 0]

    # 현재 위치와 방향 
    r, c, d = 0, 0, 0 

    for num in range(1, N*N+1):
        arr[r][c] = num 

        nr, nc = r + dr[d], c + dc[d]

        if not (0 <= nr < N and 0 <= nc < N) or arr[nr][nc] != 0:
            d = (d + 1) % 4
            nr, nc = r + dr[d], c + dc[d]
        
        r, c = nr, nc 
    
    print(f'#{t}')
    for row in arr:
        print(*row)
```
