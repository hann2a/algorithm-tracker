## CODE

```python

import sys
sys.stdin = open("input.txt", "r")

T = 10
for _ in range(T):
    tc = int(input())
    grid = [list(map(int, input().split())) for _ in range(100)]

    # 2가 있는 시작 열 찾기 (맨 아래 행)
    r = 99
    c = grid[99].index(2)

    while r > 0:
        # 왼쪽으로 갈 수 있으면 끝까지 이동
        if c - 1 >= 0 and grid[r][c - 1] == 1:
            while c - 1 >= 0 and grid[r][c - 1] == 1:
                c -= 1
            r -= 1

        # 오른쪽으로 갈 수 있으면 끝까지 이동
        elif c + 1 < 100 and grid[r][c + 1] == 1:
            while c + 1 < 100 and grid[r][c + 1] == 1:
                c += 1
            r -= 1

        # 좌우가 막혀있으면 그냥 위로
        else:
            r -= 1

    print(f"#{tc} {c}")
    ```
