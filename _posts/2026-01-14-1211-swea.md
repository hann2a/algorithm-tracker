## CODE

```python

import sys

sys.stdin = open('input.txt', 'r')

T = 10

def measure_lenth(arr, startpoint):
    x, y = 0, startpoint
    count = 0  # 좌우 이동 횟수

    while x < 99:
        if y - 1 >= 0 and arr[x][y-1] == 1:
            while y - 1 >= 0 and arr[x][y-1] == 1:
                y -= 1
                count += 1  # 왼쪽 이동 카운트

        elif y + 1 < 100 and arr[x][y+1] == 1:
            while y + 1 < 100 and arr[x][y+1] == 1:
                y += 1
                count += 1  # 오른쪽 이동 카운트

        x += 1  # 매 반복마다 무조건 아래로

    return count


for t in range(1, T+1):
    num = int(input())
    arr = [list(map(int, input().split())) for _ in range(100)]

    shortest_long = float('inf')
    result_y = 0  # 가장 짧은 경로의 시작 위치

    for i in range(100):
        if arr[0][i] == 1:
            length = measure_lenth(arr, i)
            if length < shortest_long:
                shortest_long = length
                result_y = i  # 시작 위치 저장

    print(f'#{t} {result_y}')
```
